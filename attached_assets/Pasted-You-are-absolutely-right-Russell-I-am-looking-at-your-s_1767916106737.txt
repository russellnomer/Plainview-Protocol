# --- HELPERS ---
@st.cache_data
def get_reps(state_full_name):
    """
    Fetches real legislators from UnitedStates.io (Open Data).
    Converts state name (New York) to code (NY) and filters.
    """
    # 1. State Name to Abbreviation Mapping
    us_state_to_abbrev = {
        "Alabama": "AL", "Alaska": "AK", "Arizona": "AZ", "Arkansas": "AR", "California": "CA",
        "Colorado": "CO", "Connecticut": "CT", "Delaware": "DE", "Florida": "FL", "Georgia": "GA",
        "Hawaii": "HI", "Idaho": "ID", "Illinois": "IL", "Indiana": "IN", "Iowa": "IA",
        "Kansas": "KS", "Kentucky": "KY", "Louisiana": "LA", "Maine": "ME", "Maryland": "MD",
        "Massachusetts": "MA", "Michigan": "MI", "Minnesota": "MN", "Mississippi": "MS", "Missouri": "MO",
        "Montana": "MT", "Nebraska": "NE", "Nevada": "NV", "New Hampshire": "NH", "New Jersey": "NJ",
        "New Mexico": "NM", "New York": "NY", "North Carolina": "NC", "North Dakota": "ND", "Ohio": "OH",
        "Oklahoma": "OK", "Oregon": "OR", "Pennsylvania": "PA", "Rhode Island": "RI", "South Carolina": "SC",
        "South Dakota": "SD", "Tennessee": "TN", "Texas": "TX", "Utah": "UT", "Vermont": "VT",
        "Virginia": "VA", "Washington": "WA", "West Virginia": "WV", "Wisconsin": "WI", "Wyoming": "WY"
    }

    # 2. Get the code (e.g., "NY")
    state_code = us_state_to_abbrev.get(state_full_name)
    if not state_code:
        return pd.DataFrame([{"Error": "Select a valid state"}])

    # 3. Fetch Live Data from UnitedStates.io
    try:
        url = "https://theunitedstates.io/congress-legislators/legislators-current.json"
        response = requests.get(url)
        data = response.json()
        
        reps_list = []
        for person in data:
            # Check if they are currently in office and match the state
            current_term = person['terms'][-1]
            if current_term['state'] == state_code:
                # Format Name
                full_name = f"{person['name']['first']} {person['name']['last']}"
                # Format Role (Sen vs Rep)
                role = "Senator" if current_term['type'] == 'sen' else f"Rep (Dist {current_term.get('district', 'At-Large')})"
                party = current_term['party']
                
                reps_list.append({
                    "Name": full_name,
                    "Role": role,
                    "Party": party
                })
        
        # 4. Create DataFrame
        if not reps_list:
            return pd.DataFrame([{"Status": "No reps found (Check State selection)"}])
            
        df = pd.DataFrame(reps_list)
        return df

    except Exception as e:
        return pd.DataFrame([{"Error": f"Data fetch failed: {str(e)}"}])
